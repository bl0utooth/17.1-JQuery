html
<!DOCTYPE html>
<html lang="en">
<head>
  <meta charset="UTF-8">
  <title>Movie Ratings App</title>
  <script src="https://code.jquery.com/jquery-3.3.1.js"></script>
  <style>
    .movie-container {
      margin-top: 20px;
    }
  </style>
</head>
<body>

<div id="app">
  <form id="movieForm">
    <label for="title">Title:</label>
    <input type="text" id="title" required minlength="2">
    
    <label for="rating">Rating:</label>
    <input type="number" id="rating" min="0" max="10" required>

    <button type="submit">Add Movie</button>
  </form>

  <div id="movieList" class="movie-container">
    <!-- Movie entries will be appended here -->
  </div>

  <div>
    <label>Sort By:</label>
    <select id="sortType">
      <option value="titleAsc">Title (A-Z)</option>
      <option value="titleDesc">Title (Z-A)</option>
      <option value="ratingAsc">Rating (Low to High)</option>
      <option value="ratingDesc">Rating (High to Low)</option>
    </select>
  </div>
</div>

<script>
  $(document).ready(function() {
    $('#movieForm').submit(function(event) {
      event.preventDefault();

      // Capture values from the form
      var title = $('#title').val();
      var rating = $('#rating').val();

      // Validate title and rating
      if (title.length < 2 || rating < 0 || rating > 10) {
        alert("Please enter a valid title (at least 2 characters) and a valid rating (between 0 and 10).");
        return;
      }

      // Append values to the DOM
      var movieEntry = $('<div class="movie-entry">').html('<strong>Title:</strong> ' + title + ', <strong>Rating:</strong> ' + rating);
      var removeButton = $('<button class="remove-btn">Remove</button>');

      // Add click event listener to remove the entry when clicked
      removeButton.click(function() {
        movieEntry.remove();
      });

      // Append the movie entry and remove button to the movie list container
      movieEntry.append(removeButton);
      $('#movieList').append(movieEntry);

      // Clear the form inputs
      $('#title').val('');
      $('#rating').val('');
    });

    // Function to sort movie entries based on selected criteria
    function sortMovies() {
      var sortType = $('#sortType').val();
      var moviesContainer = $('#movieList');

      var movies = moviesContainer.children('.movie-entry').get();

      movies.sort(function(a, b) {
        var keyA, keyB;

        if (sortType === 'titleAsc' || sortType === 'titleDesc') {
          keyA = $(a).find('strong:first').text();
          keyB = $(b).find('strong:first').text();
        } else {
          keyA = parseFloat($(a).find('strong:last').text().replace('Rating:', '').trim());
          keyB = parseFloat($(b).find('strong:last').text().replace('Rating:', '').trim());
        }

        if (sortType.includes('Asc')) {
          return keyA.localeCompare(keyB);
        } else {
          return keyB.localeCompare(keyA);
        }
      });

      // Clear the container and append the sorted movies
      moviesContainer.empty();
      $.each(movies, function(index, movie) {
        moviesContainer.append(movie);
      });
    }

    // Call sortMovies when the sortType changes
    $('#sortType').change(function() {
      sortMovies();
    });
  });
</script>

</body>
</html>

In this version:

  • The title input has a minlength attribute to enforce a minimum length of 2 characters.
  • The rating input has min and max attributes to enforce a range between 0 and 10.
  • A dropdown menu is added to allow users to select sorting options.
  • The sortMovies function is introduced to handle the sorting logic.
  • The sorting functionality is called when the form is submitted or when the sorting dropdown is changed.
